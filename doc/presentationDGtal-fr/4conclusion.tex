\section{Conclusion et Milestones}

\begin{frame}
\frametitle{Conclusion}



\begin{block}{Points positifs\HH}
  
  \begin{itemize}
  \item Multi-site
  \item Design générique, extensible, multiplateforme
  \item Infrastructure et algorithmes de base performants
  \item Mécanisme de visualisation vectorielle 2D  pratique
  \end{itemize}
\end{block}


\begin{exampleblock}{Perspective\HH}
  
  \begin{itemize}
  \item Intégrer de nouveaux sites/développeurs
  \item Intégrer de nouveaux algorithmes
  \item Construire une documentation orientée utilisateurs
  \item Poursuivre la construction de ``shortcuts''
  \item Créer des démonstrateurs 
  \item \ldots
  \end{itemize}
\end{exampleblock}

\end{frame}

\begin{frame}
\frametitle{Versions futures}

0.3
\begin{itemize}
\item Analyse volumétrique en distance complète (RDT, MA)
\item Espace de Khalimsky (modèle inter-pixel) et operateurs de bord
\item Couverture tangentielle et estimateurs géométriques 2D
\end{itemize}

0.4
\begin{itemize}
\item Conteneurs d'image avancés (tuilés, backends VTK-ITK,...)
\item Modèle topologique de partitions (ex. carte combinatoire)
\item Plus de primitives géométriques
\item Analyse surfacique en dimension 3 et plus
\end{itemize}

mais tout dépend de vous...
